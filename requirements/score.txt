# scoring env (single, shared by all benchmarks) — evaluate.py runs the vendored
# benchmarks/scorers/spatialscore/{evaluate_results.py, utils/util.py}.
# This is a superset of every scorer's deps: spatialscore needs the heavy vllm/torch
# stack below, and refspatial_expand/multihopspatial only need numpy/pillow/PyYAML
# (already pinned here), so one env covers all of them.
# Rebuilt FROM SCRATCH (see README). Pinned to the working simon_sscore (Python 3.10, CUDA 12.8).
#   conda create -n <name> python=3.10 -y && conda activate <name>
#   pip install -r requirements/score.txt
#
# The --extra-index-url pulls the cu128 torch/torchvision builds; vllm 0.11.0 is
# compatible with torch 2.8.0. The judge LLM (default openai/gpt-oss-20b) is resolved
# from the HF cache at run time (SS_LLM_PATH to override) — not a pip dependency, but it
# needs openai-harmony (its chat format) below.

--extra-index-url https://download.pytorch.org/whl/cu128

vllm==0.11.0            # judge engine (evaluate_results.py imports it); pulls transformers/triton/xformers
transformers>=4.55.2,<5 # ★ CAP: vllm 0.11.0 only says >=4.55.2 (no upper bound), so pip drifts to
                        # transformers 5.x, which drops tokenizer.all_special_tokens_extended and
                        # breaks vllm's tokenizer proxy ("TokenizersBackend has no attribute ...").
                        # 4.55+ keeps gpt-oss (GptOssForCausalLM) support; <5 keeps the old API.
torch==2.8.0            # +cu128 via the index above
torchvision==0.23.0     # util.py imports it (+cu128)
matplotlib==3.10.9      # util.py imports it
numpy==2.2.6
pillow==12.2.0
tqdm==4.68.3
PyYAML==6.0.3
openai-harmony==0.0.8   # gpt-oss-20b harmony chat format (Stage-2 judge)
